void PostOrderTraversal(BinTree BT)
{
if(!BT) return;
BinTree T=BT;
BinTree Stack[maxsize];int top=-1;

while(top!=-1||T)
{
    while(T)
    {
        Stack[++top]=T;
        T=T->left;
    }

    T=Stack[top];
    T->times++;
    top--;

    if(T->times==1)
    {
        Stack[++top]=T;
        T=T->right;
    }
    else 
    {
        printf("%d",T->data);
        T=NULL;
    }

}
